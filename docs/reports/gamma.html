IID Gamma simulation

This report summarizes the Monte Carlo simulation study for the IID Gamma goodness-of-fit test implemented in gofedf::testGamma().

Data are generated from Gamma distributions with scale equal to 1 and shape parameters 1, 7, and 50. For each setting, we examine the distribution of p-values and estimate the empirical type-I error rate at the nominal level 0.05.

sample_sizes <- c(25, 50, 100, 250)
shape_values <- c(1, 7, 50)
discretize_values <- c(FALSE, TRUE)
alpha <- 0.05

Read p-values

read_gamma_pvalues <- function(n, shape, discretize) {

  file <- paste0(
    "../results/gamma/gamma_iid_n", n,
    "_shape", shape,
    "_discretize_", discretize,
    "_pvalues.rds"
  )

  if (!file.exists(file)) {
    stop("File does not exist: ", file)
  }

  pvals <- readRDS(file)

  tibble(
    n = n,
    shape = shape,
    scale = 1,
    discretize = discretize,
    rep = seq_along(pvals),
    cvm_pvalue = map_dbl(pvals, 1),
    ad_pvalue = map_dbl(pvals, 2)
  )
}
gamma_results <- expand.grid(
  n = sample_sizes,
  shape = shape_values,
  discretize = discretize_values
) |>
  pmap_dfr(function(n, shape, discretize) {
    read_gamma_pvalues(
      n = n,
      shape = shape,
      discretize = discretize
    )
  })

gamma_long <- gamma_results |>
  pivot_longer(
    cols = c(cvm_pvalue, ad_pvalue),
    names_to = "method",
    values_to = "pvalue"
  ) |>
  mutate(
    method = recode(
      method,
      cvm_pvalue = "CvM",
      ad_pvalue = "AD"
    )
  )

Type-I error rate

type_one_table <- gamma_long |>
  group_by(n, shape, scale, method, discretize) |>
  summarise(
    MC = n(),
    type_one_error = mean(pvalue < alpha, na.rm = TRUE),
    .groups = "drop"
  )
